109 vector<T> dijkstra(const T &start, const T &end) const {
110 // Priority queue to store vertices with their distances
111 priority_queue<pair<double, Vertex<T>*>, vector<pair<double, Vertex<T>*>>, greater<pair<double, Vertex<T>*>>> pq;
112
113 // Map to store the distance from the start vertex to each vertex
114 unordered_map<Vertex<T>*, double> distance;
115
116 // Map to store the previous vertex in the shortest path
117 unordered_map<Vertex<T>*, Vertex<T>*> previous;
118
119 // Initialize distances and add the start vertex to the priority queue
120 for (auto v : vertexSet) {
121 distance[v] = numeric_limits<double>::infinity();
122 previous[v] = nullptr;
123 }
124
125 Vertex<T> *startVertex = findVertex(start);
126 Vertex<T> *endVertex = findVertex(end);
127
128 if (!startVertex || !endVertex) {
129 cerr << "Start or end vertex not found." << endl;
130 return vector<T>();
131 }
132
133 distance[startVertex] = 0.0;
134 pq.push({0.0, startVertex});
135
136 while (!pq.empty()) {
137 Vertex<T>* currentVertex = pq.top().second;
138 pq.pop();
139
140 if (currentVertex == endVertex) {
141 // Reconstruct the path from end to start
142 vector<T> path;
143 while (currentVertex != nullptr) {
144 path.insert(path.begin(), currentVertex->getInfo());
145 currentVertex = previous[currentVertex];
146 }
147 return path;
148 }
149
150 for (const Edge<T>& edge : currentVertex->getAdj()) {
151 Vertex<T>* neighbor = edge.getDest();
152 double newDistance = distance[currentVertex] + edge.getWeight();
153
154 if (newDistance < distance[neighbor]) {
155 distance[neighbor] = newDistance;
156 previous[neighbor] = currentVertex;
157 pq.push({newDistance, neighbor});
158 }
159 }
160 }
161 return vector<T>();
162 }

502 vector<T> res;

504 if (s == NULL)
505 return res;
506 queue<Vertex<T> *> q;
507 for (auto v : vertexSet)
508 v->visited = false;
509 q.push(s);
510 s->visited = true;
511 while (!q.empty()) {
512 auto v = q.front();
513 q.pop();
514 res.push_back(v->info);
515 for (auto & e : v->adj) {

570template<class T>
571vector<T> Graph<T>::topsort() const {
572 vector<T> res;
573 for (auto w: vertexSet) {
574 w->indegree = 0;
575 }
576
577 for (auto v: vertexSet) {
578 for (auto& e : v->adj) {
579 e.dest->indegree++;
580 }
581 }
582
583 queue<Vertex<T>*> q;
584 for (auto v: vertexSet) {
585 if (v->indegree == 0) {
586 q.push(v);
587 }
588 }
589 while(!q.empty()) {
590 Vertex<T> * v = q.front();
591 q.pop();
592 res.push_back(v->info);
593 for (auto& e : v->adj) {
594 auto w = e.dest;
595 w->indegree--;
596 if (w->indegree == 0) {
597 q.push(w);
598 }
599 }
600 }
601 if (res.size() != vertexSet.size()) {
602 res.clear();
603 return res;
604 }
605 return res;
